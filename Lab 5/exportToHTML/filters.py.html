4    def grayscale(image): 
5        """ (Cimpl.Image) -> None 
6         
7        Convert image into shades of gray. 
8         
9        >>> image = load_image(choose_file())  
10       >>> grayscale(image) 
11       >>> show(image)     
12       """ 
13       for x, y, (r, g, b) in image: 
14    
15           # Use the shade of gray that has the same brightness as the pixel's 
16           # original color. 
17            
18           brightness = (r + g + b) // 3 
19           gray = create_color(brightness, brightness, brightness) 
20           set_color(image, x, y, gray)

41   def solarize(image, threshold): 
42       """ (Cimpl.Image, int) -> None 
43       Solarize image, modifying the RGB components that 
44       have intensities that are less than threshold. 
45       Parameter threshold is in the range 0 to 256, inclusive. 
46       >>> image = load_image(choose_file()) 
47       >>> solarize(image, int(input())) 
48       >>> show(image) 
49       """ 
50       for x, y, (r, g, b) in image: 
51    
52           if r < threshold: 
53               r = 255 - r 
54    
55           if g < threshold: 
56               g = 255 - g 
57    
58           if b < threshold: 
59               b = 255 - b 
60    
61           solarized = create_color(r, g, b) 
62           set_color(image, x, y, solarized)

75   def black_and_white(image): 
76       """ (Cimpl.Image) -> None 
77        
78       Convert image to a black-and-white (two-tone) image. 
79        
80       >>> image = load_image(choose_file())  
81       >>> black_and_white(image) 
82       >>> show(image)      
83       """ 
84       black = create_color(0, 0, 0) 
85       white = create_color(255, 255, 255) 
86        
87       # Brightness levels range from 0 to 255. 
88       # Change the colour of each pixel to black or white, depending on  
89       # whether its brightness is in the lower or upper half of this range.        
90    
91       for x, y, (r, g, b) in image: 
92           brightness = (r + g + b) // 3 
93            
94           if brightness < 128: 
95               set_color(image, x, y, black) 
96           else:     # brightness is between 128 and 255, inclusive 
97               set_color(image, x, y, white)

124  def black_and_white_and_gray(image): 
125      """ (Cimpl.Image) -> None 
126       
127      Convert image to a black-and-white-and-gray (three-tone) image. 
128   
129      >>> image = load_image(choose_file())  
130      >>> black_and_white_and_gray(image) 
131      >>> show(image)      
132      """ 
133      black = create_color(0, 0, 0) 
134      gray = create_color(128, 128, 128) 
135      white = create_color(255, 255, 255) 
136   
137      # Brightness levels range from 0 to 255. Change the colours of 
138      # pixels whose brightness is in the lower third of this range to black, 
139      # in the upper third to white, and in the middle third to medium-gray. 
140   
141      for x, y, (r, g, b) in image: 
142          brightness = (r + g + b) // 3 
143   
144          if brightness < 85: 
145              set_color(image, x, y, black) 
146          elif brightness < 171:  # brightness is between 85 and 170, inclusive 
147              set_color(image, x, y, gray) 
148          else:                  # brightness is between 171 and 255, inclusive 
149              set_color(image, x, y, white)

152  def sepia_tint(image): 
153      """ (Cimpl.Image) -> None 
154      Convert image to sepia tones. 
155      >>> image = load_image(choose_file()) 
156      >>> sepia_tint(image) 
157      >>> show(image) 
158      """ 
159      grayscale(image) 
160      for x, y, (r, g, b) in image: 
161          if r < 63: 
162              b = b * 0.9 
163              r = r * 1.1 
164          elif r < 192: 
165              b = b * 0.85 
166              r = r * 1.15 
167          else: 
168              b = b * 0.93 
169              r = r * 1.08 
170          set_color(image, x, y, create_color(r, g, b))

173  def _adjust_component(amount): 
174      """ (int) -> int 
175      Divide the range 0..255 into 4 equal-size quadrants, 
176      and return the midpoint of the quadrant in which the 
177      specified amount lies. 
178      >>> _adjust_component(10) 
179      31 
180      >>> _adjust_component(85) 
181      95 
182      >>> _adjust_component(142) 
183      159 
184      >>> _adjust_component(230) 
185      223 
186      """ 
187      if amount <= 63: 
188          return 31 
189      elif amount <= 127: 
190          return 95 
191      elif amount <= 191: 
192          return 159 
193      else: 
194          return 223

208  def _adjust_component_test(amount, n): 
209      """ 
210      >>> _adjust_component_test(10,4) 
211      31 
212      >>> _adjust_component_test(85,4) 
213      95 
214      >>> _adjust_component_test(142,4) 
215      159 
216      >>> _adjust_component_test(230,4) 
217      223 
218      """ 
219      for i in range(0, n): 
220          if amount <= 255//n + 255//n * i: 
221              return int(255 / n + 255 / n * (i - 1) + 255 / n / 2)
