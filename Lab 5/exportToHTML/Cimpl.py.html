60   def _adjust_component(comp): 
61       """Return comp as an integer between 0 and 255, inclusive, returning 0 
62       if comp is negative and capping values >= 256 at 255. 
63       """ 
64       comp = int(comp) 
65       return max(0, min(255, comp))

67   class Color(tuple): 
68       """An RGB color. 
69    
70       When an instance is created, the RGB component values are quietly adjusted, 
71       as required, to ensure that they are ints in the range 0..255, inclusive. 
72    
73       Examples: 
74         Color(120, 60, 200) yields the color (120, 60, 200) 
75         Color(-120, 60, 280) yields the color (0, 60, 255) 
76         Color(120.0, 60.5, 200.2) yields the color (120, 60, 200) 
77    
78       Because Color is a subclass of tuple, Color objects can be treated as 
79       tuples. For example, to retrieve the rgb components stored in a Color 
80       object, it can be subscripted (indexed): 
81    
82         col = Color(120, 60, 200) 
83         ... 
84         r = col[0]  # r is bound to 120 
85         g = col[1]  # g is bound to 60 
86         b = col[2]  # b is bound to 200 
87    
88       Or, we can unpack a Color object, the same way we can unpack a tuple: 
89    
90         r, g, b = col  # r is bound to 120, g is bound to 60, b is bound to 200 
91    
92       To convert a Color object col to a tuple, do this: 
93    
94         tuple(col)  # Returns the tuple (120, 60, 200) 
95       """ 
96       __slots__ = () # Binding __slots__ to an empty tuple prevents instance 
97                      # dictionaries from being created, reducing memory 
98                      # requirements 
99    
100      def __new__(_cls, red, green, blue): 
101          """Return a new instance of Color(red, green, blue).""" 
102          return tuple.__new__(_cls, (_adjust_component(red), 
103                                      _adjust_component(green), 
104                                      _adjust_component(blue))) 
105   
106      @classmethod 
107      def _make(cls, t): 
108          # Make a new Color object from (r, g, b) tuple t. 
109          # This method assumes that t is a 3-tuple of ints, each in the 
110          # range 0 to 255, inclusive. THIS IS NOT CHECKED. 
111          # 
112          # The preferred way for application code to convert a tuple t to a 
113          # Color object is: 
114          # 
115          #     col = Color(t[0], [t1], t[2]) 
116          # 
117          # or: 
118          # 
119          #     r, g, b = t 
120          #     col = Color(r, g, b) 
121          # 
122          # This way ensures that the rgb components will be converted, if 
123          # necessary, to integers in the range 0..255 when the Color object is 
124          # initialized. 
125   
126          # Originally, I was going to have this function verify that it 
127          # was passed a 3-tuple. 
128          # 
129          # if not isinstance(t, tuple): 
130          #     raise TypeError('Argument is not a tuple') 
131          # if len(t) != 3: 
132          #     raise TypeError('Expected 3 values in tuple, got %d' % len(t)) 
133          return tuple.__new__(cls, t) 
134   
135      def __repr__(self): 
136          """Return the "official" string representation of the Color. 
137   
138          This string is a valid expression that will yield a Color object with 
139          the same value when passed to eval(). 
140          """ 
141          return 'Color(red={0[0]}, green={0[1]}, blue={0[2]})'.format(self)

144  class Image(object): 
145      """ 
146      A Image is a wrapper for an instance of PIL's Image class. 
147      Supported image formats include: JPEG, GIF, TIFF, PNG and BMP. 
148   
149      To load an image from a file: 
150   
151         image = Image(a_filename) 
152   
153      To create a blank image with specified dimensions: 
154   
155          image = Image(width=width_in_pixels, height=height_in_pixels) 
156   
157      By default, the blank image's color is white. A different image color can be 
158      specified with a Color object: 
159   
160          image = Image(width=width_in_pixels, height=height_in_pixels 
161                        color=Cimpl.Color(red, green, blue)) 
162   
163      To duplicate an image: 
164   
165          original = Image(...) 
166          duplicate = Image(image=original) 
167      """ 
168   
169      def __init__(self, filename=None, image=None, 
170                   width=None, height=None, color=Color(255, 255, 255)): 
171   
172          if filename is not None: # load image from file 
173              self.pil_image = PIL.Image.open(filename).convert("RGB") 
174              self.filename = filename 
175   
176          elif image is not None:  # copy an image 
177              # To make a deep copy of the Image we're duplicating, we need to 
178              # copy its PIL Image. 
179              self.pil_image = image.pil_image.copy() 
180              self.filename = None 
181   
182          elif width is None and height is None and color is None: 
183              raise TypeError('Image(): called with no arguments?') 
184          elif width is None or height is None: 
185              raise TypeError('Image(): missing width or height argument') 
186          elif width > 0 and height  > 0:  # create a blank image 
187              self.pil_image = PIL.Image.new(mode="RGB", size=(width, height), 
188                                             color=tuple(color)) 
189              self.filename = None 
190          else: 
191              raise ValueError('Image(): width and height must be > 0') 
192   
193          self.zoomfactor = 1 # By default, display images at their 
194                              # original size. 
195          self.pixels = self.pil_image.load() # The pixel access object for the 
196                                              # PIL Image; essentially a 2-D array 
197                                              # of (r, g, b) tuples. 
198   
199      def copy(self): 
200          """Return a deep copy of this Image. 
201          """ 
202          dup = Image(image=self) 
203          return dup 
204   
205      def set_zoom(self, factor): 
206          """Specify the amount that the image should be expanded when it is 
207          displayed; e.g., if factor is 3 the image is displayed at 
208          3 times its original size. 
209          """ 
210          if isinstance(factor, int) and factor > 0: 
211              self.zoomfactor = factor 
212          else: 
213              raise ValueError("factor must be a positive integer") 
214   
215      def get_width(self): 
216          """Return the width of this Image, in pixels. 
217          """ 
218          return self.pil_image.size[0] 
219   
220      def get_height(self): 
221          """Return the height of this Image, in pixels. 
222          """ 
223          return self.pil_image.size[1] 
224   
225      def get_filename(self): 
226          """Return the name of the file where this Image is stored. 
227          """ 
228          return self.filename 
229   
230      def __iter__(self): 
231          """Return a generator object that iterates over this Image's pixels 
232          from left to right, top to bottom. The values when iterating are 
233          Color objects, each containing the RGB color of one pixel. 
234          """ 
235          width = self.get_width() 
236          height = self.get_height() 
237   
238          for y in range(0, height): 
239              for x in range(0, width): 
240                  col = Color._make(self.pixels[x, y]) 
241                  yield x, y, col 
242   
243      def get_color(self, x, y): 
244          """Return a Color containing the RGB components of the pixel at 
245          location (x, y) in this Image. 
246          """ 
247          return Color._make(self.pixels[x, y]) 
248   
249      def set_color(self, x, y, color): 
250          """Set the color of the pixel at location (x, y) in this Image, 
251          to the RGB values stored in Color object, color. 
252          """ 
253          # Ensure that color is bound to a Color object before we update the 
254          # pixel access object. Calling isinstance is frowned upon in some 
255          # circles, but we need to ensure that color is not bound to an 
256          # arbitrary sequence containing values outside the range 0..255. 
257   
258          if not isinstance(color, Color): 
259              raise TypeError('Parameter color is not a Color object') 
260           
261          #self.pixels[x, y] = (color[0], color[1], color[2]) 
262          self.pixels[x, y] = tuple(color) 
263   
264      def write_to(self, filename): 
265          """Save this Image to filename, overwriting the existing file. 
266   
267          Raise a ValueError if 
268           - filename is None; 
269           - if filename has no extension. 
270           - if the filename's extension doesn't specify an image file format 
271             supported by this module. 
272   
273          FIXME: reset the image's filename. 
274          """ 
275          if filename: 
276              ext = os.path.splitext(filename)[-1] 
277              if ext == '': 
278                  raise ValueError('Filename has no extension') 
279   
280              # Extensions must be entirely lower-case or upper-case, but not 
281              # mixed case. 
282              if ext in IMAGE_FILE_FORMATS or \ 
283                     (ext.isupper() and ext.lower() in IMAGE_FILE_FORMATS): 
284                  self.pil_image.save(filename) 
285                  #self.set_filename_and_title(filename) 
286              else: 
287                  raise ValueError("%s is not a supported image file format." \ 
288                                    % ext) 
289          else: 
290              raise ValueError("Parameter filename is None.") 
291   
292      def _zoom_image(self): 
293          """Return a copy of this Image, expanding it by the image's 
294          zoom factor (see set_zoom). 
295          """ 
296          copy = Image(width=self.get_width() * self.zoomfactor, 
297                         height=self.get_height() * self.zoomfactor, 
298                         color=Color(255, 255, 255)) 
299   
300          for x, y, col in self: 
301              scaled_x = x * self.zoomfactor 
302              scaled_y = y * self.zoomfactor 
303              for j in range(self.zoomfactor): 
304                  for i in range(self.zoomfactor): 
305                      copy.set_color(scaled_x + i, scaled_y + j, col) 
306          return copy 
307   
308      def show(self): 
309          root = Tk() 
310   
311          # By default, display this image's PIL image access object. 
312          pil_image = self.pil_image 
313   
314          if self.zoomfactor != 1: 
315              # Make an enlarged copy of this image and display its 
316              # PIL image access object. 
317              pil_image = self._zoom_image().pil_image 
318   
319          if self.filename is None: 
320              view = ImageViewer(root, pil_image) 
321          else: 
322   
323              # Use the name of the image file, without the drive/directory part 
324              # of its pathname, as the window's title. 
325              title = os.path.basename(self.filename) 
326              view = ImageViewer(root, pil_image, title) 
327   
328          root.mainloop()

334  class ImageViewer(object): 
335      def __init__(self, master, pil_image, title = "New Image"): 
336          """Initialize an image viewer (a Tk window) with parent widget master. 
337          pil_image is bound to the instance of PIL.Image.Image that contains 
338          the image to be displayed. 
339          """ 
340          master.title(title) 
341   
342          image_width = pil_image.size[0] 
343          image_height = pil_image.size[1] 
344   
345          # Build a canvas big enough to display the image 
346          self.canvas = Canvas(master, 
347                               width=image_width, 
348                               height=image_height) 
349   
350          self.photo_image = PIL.ImageTk.PhotoImage(pil_image) 
351          # The PhotoImage object must be bound to an instance variable 
352          # (which exists for the lifetime of the ImageViewer object) instead 
353          # of a local variable. If we don't do this, the PhotoImage object 
354          # might be garbage collected after __init__ returns, but before 
355          # we run the Tk/Tcl event loop, and the image won't appear in the 
356          # canvas. This is a bug in PIL... 
357   
358          # Place the image in the canvas. 
359          self.canvas.create_image(image_width // 2, 
360                                   image_height // 2, 
361                                   image = self.photo_image) 
362   
363          self.canvas.pack() 
364   
365          master.resizable(0, 0) # Don't allow the window to be resized

370  def create_color(red, green, blue): 
371      """(int, int, int) -> Cimpl.Color 
372       
373      Return a Color object with the RGB components specified by red, green 
374      and blue. 
375   
376      When the Color object is created, non-integer component values are 
377      converted, if possible, to ints; negative values are converted to 0, 
378      and values > 255 are capped at 255. 
379      """ 
380      return Color(red, green, blue) 
381   
382  def distance(color1, color2): 
383      """(Cimpl.Color, Cimpl.Color) -> float 
384       
385      Return the Euclidean distance between two Color objects. 
386      """ 
387      r1, g1, b1 = color1 
388      r2, g2, b2 = color2 
389   
390      return math.sqrt((r1 - r2) ** 2 + (g1 - g2) ** 2 + (b1 - b2) ** 2)

450  def save_as(pict, filename=None): 
451      """ (Cimpl.Image) -> None 
452          (Cimpl.Image, str) -> None 
453           
454      Save Image pict to the specified file. If no filename is supplied, 
455      first prompt the user to interactively choose a directory and 
456      filename. 
457   
458      Examples: 
459        save_as(pict, 'mypicture.jpg') saves pict to mypicture.jpg 
460        save_as(pict) asks the user to choose the directory and filename 
461      """ 
462      if not filename: 
463          # The suggested name for the file is the image's current filename, 
464          # if it has one; otherwise, use 'untitled'. 
465          if pict.get_filename(): 
466              base = os.path.basename(pict.get_filename()) 
467              initial = os.path.splitext(base)[0] 
468          else: 
469              initial = 'untitled' 
470          filename = choose_save_filename(initial) 
471   
472      if filename: 
473          pict.write_to(filename) 
474   
475  def save(pict): 
476      """ (Cimpl.Image) -> None 
477       
478      Save Image pict to its file, overwriting the existing file. 
479   
480      If this Image doesn't have a corresponding filename; i.e., this 
481      instance has not yet been written to a file, the user will be prompted 
482      to provide a filename. See save_as(pict, filename). 
483      """ 
484      name = pict.get_filename() 
485      if name: 
486          pict.write_to(name) 
487      else: 
488          save_as(pict)

519  def choose_save_filename(initial=''): 
520      """ (None) -> str 
521          (str) -> str 
522   
523      Display a Save As dialogue box. Return the complete path to  
524      the new file. 
525       
526      Parameter initial is the string that is displayed in the dialogue  
527      box's File name field. This parameter is optional; if it is not provided  
528      when the function is called, the File name field is empty. 
529      """ 
530      root = Tk() 
531      # Hide the top-level window. (We only want the Save As dialogue box 
532      # to appear.) 
533      root.withdraw() 
534   
535      path = tkinter.filedialog.asksaveasfilename(filetypes=IMAGE_FILE_TYPES, 
536                                            initialfile=initial, 
537                                            defaultextension='.jpg') 
538   
539      # Things I've discovered about the dialogue box displayed by 
540      # asksaveasfilename(): 
541      # 
542      # If the name we type in the "File name" field has no extension, 
543      # the extension corresponding to the selected "Save as type" is appended 
544      # to the name returned by the function. 
545      # An exception to this occurs when "All files" is selected as the 
546      # "Save as type" and we type a name without an extension. In this case, 
547      # defaultextension is appended to the name. 
548   
549      # We can also type a name with an extension. If the extension is listed 
550      # in the IMAGE_FILE_TYPES list, that name is returned as typed, with no 
551      # changes to the extension; in other words, the extension implied by the 
552      # selected "Save as type" isn't used. 
553   
554      # All bets are off if we type a name with an extension that isn't listed 
555      # in the IMAGE_FILE_TYPES list. Sometimes an additional extension 
556      # (corresponding to the selected "Save as type") is appended, but sometimes 
557      # this doesn't happen. I haven't found an explanation for this behaviour 
558      # in any of the online documentation or examples for Tkinter. 
559   
560      root.destroy() # Do we need to do this? 
561      return path
